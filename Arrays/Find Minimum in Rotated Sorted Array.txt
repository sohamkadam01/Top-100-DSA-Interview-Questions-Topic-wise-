class Solution {
    public int findMin(int[] nums) {
         
         approch 1

         Arrays.sort(nums);
         return nums[0];
           
         approch 2

        int left = 0;
        int right = nums.length-1;

        while(left<right){
            int mid = (left + right) / 2;

            if(nums[mid] > nums[right]){
                left = mid + 1;
            }
            else{
                right = mid;
            }
        }
        return nums[left];
    }
}
