class Solution {
    public boolean isPalindrome(String s) {
        s = s.toLowerCase();
        s = s.trim();
        StringBuilder sb = new StringBuilder();
        for(char ch : s.toCharArray()){
            if(ch >= 'a' && ch <= 'z'){
                sb.append(ch);
            }
            else if (ch == ' '){
                continue;
            }
            else if(ch >= '0' && ch <= '9'){
                sb.append(ch);
            }
        }
        String newStr = new String(sb);
        int left = 0;
        int right = newStr.length()-1;
        while(left<=right){
            if(newStr.charAt(left)==newStr.charAt(right)){
            left++;
            right--;
                continue;
            }
            else{
                return false;
            }
        }
        return true;
    }
}
